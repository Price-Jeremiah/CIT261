Javascript Object Creation Function

Three methods: Literal, New, and Construct

Literal

		var person = {firstName:"Jack", lastName:"Micheal", age:55, eyeColor:"brown", dancestyle:"fire"};

document.getElementById("demo").innerHTML =
person.firstName + " is " + person.age + " years old.";

New

		var person = new Object();
person.firstName = "John";
person.lastName = "Doe";
person.age = 50;
person.eyeColor = "blue"; 
person.dancestyle = "fire";

document.getElementById("demo").innerHTML =
person.firstName + " is " + person.age + " years old.";

Construct


function Person(first, last, age, eye, dancestyle) {
  this.firstName = first;
  this.lastName = last;
  this.age = age;
  this.eyeColor = eye;
  this.dance = dancestyle;
}

// Create a Person object
var myFather = new Person("John", "Doe", 50, "blue", "fire");

// Display age
document.getElementById("demo").innerHTML =
"My father is " + myFather.age + ".";
